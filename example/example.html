Документация

Создание объекта

var clock = new Clock({elem: $('#clock')});
$('#clock').clock();

Параметры

{
    elem        : $('#clock') /*jQuery объект (только для new)*/
    timeZone    : 4,          /*Часовой пояс*/
    autoStart   : true,       /*Автозапуск при создании*/
    timer       : false,      /*Режим таймера*/
    direction   : "donw",     /*Направление отсчета donw/up*/
    startTime   : 0,          /*С какого времени стартовать*/
    timeDif     : 120,        /*Разница во времени в секундах*/
    sRndrTpl    : "hh;:;min", /*Шаблон рендера*/
    renderAnim  : false,      /*Использовать ли анимацию*/
    renderType  : "num",      /*Тип рендера num/img*/
    frameHeight : 32,         /*Высота кадра спрайта*/
    btns : {
        play    : $('#clock_play'), /*jQuery объект кнопки play*/
        pause   : $('#clock_pause'),/*jQuery объект кнопки pause*/
        stop    : $('#clock_stop')  /*jQuery объект кнопки stop*/
    }
}

Методы

    play();         //запускае часы
    pause();        //ставит часы на паузу
    stop();         //останавливает часы
    getSurplus();   //возвращает остаток времени
    destroy();      //удаляет часы
    destroyHTML();  //удаляет часы и html

Вызов методов

    $('#clock').clock('destroy');   //возращается $('#clock')
    $('#clock').clock('getSurplus');
    //вернет $('#clock') вместо числа
    $('#clock').data('clock').getSurplus();
    /*можно вызвать метод через событие*/
    $('#clock').trigger('clock.play');
    $('#clock').trigger('clock.pause');
    $('#clock').trigger('clock.stop');
    $('#clock').trigger('clock.destroy');
    $('#clock').trigger('clock.destroyHTML');

Колбеки

{
    onCreate : function(){},     //при создании
    onPlay : function(){},       //при запуске
    onStep : function(){},       //на каждом шаге
    onPause : function(){},      //при паузе    
    onStop : function(surplus){}, //при стопе
    onDestroy : function(){},    //при удалении
    onDestroyHTML : function(){} //при удалении с html
}

События

Можно подписаться на события, если вы не передаете колбек.

$('#clock').on('clock.onStop', function(e,surplus){
    console.log(surplus);
});

Рендер

Можно передать свой рендер, который будет вызываться каждую секунду.
В качестве параметра передается текущая дата. Для получения корректной даты нужно использовать UTC методы, например getUTCSeconds.

{
    render : function(data){},  //функция рендера
}
